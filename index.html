const USERS_SHEET_NAME = 'Users'; const TICKETS_SHEET_NAME = 'Tickets'; const STORES_SHEET_NAME = 'Stores'; const SESSION_SECONDS = 21600; function doGet(e) { try { const p = e && e.parameter ? e.parameter : {}; const callback = safeCallback(p.callback || 'ticketingCallback'); const action = String(p.action || 'login').trim(); let result; if (action === 'tickets' || action === 'dashboard') { result = getManagerTickets(p.token || ''); } else if (action === 'store') { result = getStoreByCode(p.code || '', p.token || ''); } else { result = authenticateUser({ username: p.username || '', password: p.password || '' }); } return ContentService .createTextOutput(callback + '(' + JSON.stringify(result) + ');') .setMimeType(ContentService.MimeType.JAVASCRIPT); } catch (error) { return ContentService .createTextOutput('ticketingCallback(' + JSON.stringify({ success: false, message: 'خطا در ارتباط با سرور', error: String(error.message || error) }) + ');') .setMimeType(ContentService.MimeType.JAVASCRIPT); } } function doPost(e) { try { const data = JSON.parse((e && e.postData && e.postData.contents) || '{}'); if (data.action === 'login') { return jsonResponse(authenticateUser(data)); } if (data.action === 'updateTicket') { return jsonResponse(updateTicket(data)); } return jsonResponse(saveTicket(data)); } catch (error) { return jsonResponse({ success: false, message: 'خطا در پردازش اطلاعات', error: String(error.message || error) }); } } function authenticateUser(data) { const username = normalizeDigits(data.username || '').trim(); const password = normalizeDigits(data.password || '').trim(); if (!username || !password) { return { success: false, message: 'نام کاربری و رمز عبور را وارد کنید' }; } if (username === 'manager' && password === '140504') { const user = { username: 'manager', name: 'مدیر فنی', role: 'manager' }; return Object.assign({ success: true, token: createSession(user) }, user); } const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(USERS_SHEET_NAME); if (!sheet) return { success: false, message: 'شیت Users پیدا نشد' }; const rows = sheet.getDataRange().getDisplayValues(); const headerIndex = findHeaderRow(rows); if (headerIndex === -1) return { success: false, message: 'عنوان ستون‌های Users پیدا نشد' }; const headers = rows[headerIndex].map(v => String(v || '').trim()); const uCol = findColumn(headers, ['نام کاربری','کد پرسنلی','username']); const nCol = findColumn(headers, ['نام کامل','نام و نام خانوادگی','نام','name']); const pCol = findColumn(headers, ['رمز','رمز عبور','password']); const rCol = findColumn(headers, ['نقش','role']); const sCol = findColumn(headers, ['وضعیت','فعال','status']); if (uCol < 0 || nCol < 0 || pCol < 0) { return { success: false, message: 'ستون‌های Users کامل نیست' }; } for (let i = headerIndex + 1; i < rows.length; i++) { const row = rows[i]; if (normalizeDigits(row[uCol] || '').trim() === username && normalizeDigits(row[pCol] || '').trim() === password) { const status = sCol >= 0 ? String(row[sCol] || 'فعال') : 'فعال'; if (!isActiveStatus(status)) return { success: false, message: 'این حساب غیرفعال است' }; const user = { username: username, name: String(row[nCol] || '').trim(), role: rCol >= 0 && String(row[rCol] || '').trim() === 'manager' ? 'manager' : 'technician' }; return Object.assign({ success: true, token: createSession(user) }, user); } } return { success: false, message: 'نام کاربری یا رمز عبور اشتباه است' }; } function createSession(user) { const token = Utilities.getUuid().replace(/-/g, '') + Utilities.getUuid().replace(/-/g, ''); CacheService.getScriptCache().put('session:' + token, JSON.stringify(user), SESSION_SECONDS); return token; } function getSession(token) { if (!token) return null; const raw = CacheService.getScriptCache().get('session:' + token); if (!raw) return null; try { return JSON.parse(raw); } catch (e) { return null; } } function getStoreByCode(code, token) { const session = getSession(token); if (!session) { return { success: false, sessionExpired: true, message: 'نشست کاربر منقضی شده؛ دوباره وارد شوید' }; } const normalizedCode = normalizeStoreCode(code); if (!normalizedCode) return { success: false, message: 'کد فروشگاه را وارد کنید' }; const cache = CacheService.getScriptCache(); const cacheKey = 'store:' + normalizedCode; const cached = cache.get(cacheKey); if (cached) { try { return { success: true, store: JSON.parse(cached) }; } catch (e) {} } const store = findStoreByCode(normalizedCode); if (!store) return { success: false, notFound: true, message: 'کد فروشگاه در فهرست پیدا نشد' }; cache.put(cacheKey, JSON.stringify(store), SESSION_SECONDS); return { success: true, store: store }; } function getManagerTickets(token) { const session = getSession(token); if (!session || session.role !== 'manager') { return { success: false, sessionExpired: true, message: 'نشست مدیر منقضی شده؛ دوباره وارد شوید' }; } const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(TICKETS_SHEET_NAME); if (!sheet) return { success: false, message: 'شیت Tickets پیدا نشد' }; const storeMap = getStoreMap(); const values = sheet.getDataRange().getDisplayValues(); const hasHeader = values.length && String(values[0][0] || '').trim().indexOf('تاریخ') !== -1; const start = hasHeader ? 1 : 0; const tickets = []; for (let i = start; i < values.length; i++) { const r = values[i]; if (!r.some(v => String(v || '').trim())) continue; const storeCode = normalizeStoreCode(r[2] || ''); const mappedStore = storeMap[storeCode] || null; tickets.push({ id: String(r[13] || ('sheet-' + (i + 1))), rowNumber: i + 1, dateText: String(r[0] || ''), timeText: String(r[1] || ''), storeCode: storeCode, technician: String(r[3] || ''), type: String(r[4] || ''), level1: String(r[5] || ''), level2: splitPersianList(r[6]), otherText: String(r[7] || ''), consumedParts: parseParts(r[8]), hotParts: parseParts(r[9]), outsourceReason: String(r[10] || ''), reviewStatus: String(r[11] || '') || 'pending', managerNote: String(r[12] || ''), createdAt: String(r[14] || ''), technicianCode: String(r[15] || ''), storeName: String(r[16] || '') || (mappedStore ? mappedStore.name : ''), region: String(r[17] || '') || (mappedStore ? mappedStore.region : '') }); } tickets.sort((a, b) => b.rowNumber - a.rowNumber); return { success: true, tickets: tickets, updatedAt: new Date().toISOString() }; } function saveTicket(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(TICKETS_SHEET_NAME); if (!sheet) return { success: false, message: 'شیت Tickets پیدا نشد' }; if (!data.storeCode || !data.technician) return { success: false, message: 'اطلاعات تیکت کامل نیست' }; const store = findStoreByCode(data.storeCode); if (!store) return { success: false, message: 'کد فروشگاه معتبر نیست' }; ensureExtraHeaders(sheet); const session = getSession(data.token || ''); const technicianName = session && session.role === 'technician' ? session.name : String(data.technician || '').trim(); const technicianCode = session && session.role === 'technician' ? session.username : String(data.technicianCode || data.technicianUsername || '').trim(); const lock = LockService.getScriptLock(); try { lock.waitLock(10000); sheet.appendRow([ data.date || '', data.time || '', store.code, technicianName, data.type || '', data.level1 || '', arrayToText(data.level2), data.otherText || '', partsToText(data.consumedParts), partsToText(data.hotParts), data.outsourceReason || '', data.reviewStatus || 'pending', data.managerNote || '', data.id || Utilities.getUuid(), data.createdAt || new Date().toISOString(), technicianCode, store.name, store.region ]); SpreadsheetApp.flush(); return { success: true, message: 'تیکت ثبت شد', store: store }; } finally { try { lock.releaseLock(); } catch (e) {} } } function updateTicket(data) { const session = getSession(data.token || ''); if (!session || session.role !== 'manager') { return { success: false, sessionExpired: true, message: 'دسترسی مدیر معتبر نیست' }; } const rowNumber = Number(data.rowNumber); const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(TICKETS_SHEET_NAME); if (!sheet || !rowNumber || rowNumber < 2 || rowNumber > sheet.getLastRow()) { return { success: false, message: 'ردیف تیکت پیدا نشد' }; } ensureExtraHeaders(sheet); const current = sheet.getRange(rowNumber, 1, 1, 18).getDisplayValues()[0]; const values = [ current[0], current[1], current[2], current[3], data.type !== undefined ? data.type : current[4], data.level1 !== undefined ? data.level1 : current[5], data.level2 !== undefined ? arrayToText(data.level2) : current[6], data.otherText !== undefined ? data.otherText : current[7], data.consumedParts !== undefined ? partsToText(data.consumedParts) : current[8], data.hotParts !== undefined ? partsToText(data.hotParts) : current[9], data.outsourceReason !== undefined ? data.outsourceReason : current[10], data.reviewStatus !== undefined ? data.reviewStatus : current[11], data.managerNote !== undefined ? data.managerNote : current[12], current[13], current[14], current[15], current[16], current[17] ]; sheet.getRange(rowNumber, 1, 1, 18).setValues([values]); return { success: true, message: 'تیکت به‌روزرسانی شد' }; } function findStoreByCode(code) { const normalizedCode = normalizeStoreCode(code); if (!normalizedCode) return null; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STORES_SHEET_NAME); if (!sheet) throw new Error('شیت Stores پیدا نشد'); const values = sheet.getDataRange().getDisplayValues(); if (!values.length) return null; const headerInfo = getStoreHeaderInfo(values[0]); if (headerInfo.code < 0 || headerInfo.name < 0 || headerInfo.region < 0) { throw new Error('ستون‌های شیت Stores کامل نیست'); } for (let i = 1; i < values.length; i++) { const rowCode = normalizeStoreCode(values[i][headerInfo.code] || ''); if (rowCode === normalizedCode) { return { code: rowCode, name: String(values[i][headerInfo.name] || '').trim(), region: String(values[i][headerInfo.region] || '').trim() }; } } return null; } function getStoreMap() { const map = {}; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STORES_SHEET_NAME); if (!sheet) return map; const values = sheet.getDataRange().getDisplayValues(); if (!values.length) return map; const headerInfo = getStoreHeaderInfo(values[0]); if (headerInfo.code < 0 || headerInfo.name < 0 || headerInfo.region < 0) return map; for (let i = 1; i < values.length; i++) { const code = normalizeStoreCode(values[i][headerInfo.code] || ''); if (!code) continue; map[code] = { code: code, name: String(values[i][headerInfo.name] || '').trim(), region: String(values[i][headerInfo.region] || '').trim() }; } return map; } function getStoreHeaderInfo(headers) { const normalized = headers.map(v => String(v || '').trim()); return { region: findColumn(normalized, ['منطقه','ناحیه','region']), code: findColumn(normalized, ['کد','کد فروشگاه','storeCode','code']), name: findColumn(normalized, ['نام فروشگاه','فروشگاه','storeName','name']) }; } function ensureExtraHeaders(sheet) { if (sheet.getLastRow() < 1) return; const h = sheet.getRange(1, 1, 1, 18).getDisplayValues()[0]; const first = String(h[0] || '').trim(); const looksLikeHeader = first.indexOf('تاریخ') !== -1 || String(h[2] || '').indexOf('فروشگاه') !== -1; if (!looksLikeHeader) return; if (!h[13]) sheet.getRange(1, 14).setValue('شناسه تیکت'); if (!h[14]) sheet.getRange(1, 15).setValue('زمان استاندارد'); if (!h[15]) sheet.getRange(1, 16).setValue('کد پرسنلی'); if (!h[16]) sheet.getRange(1, 17).setValue('نام فروشگاه'); if (!h[17]) sheet.getRange(1, 18).setValue('منطقه'); } function safeCallback(v) { return String(v || '').replace(/[^\w.$]/g, '') || 'ticketingCallback'; } function findHeaderRow(rows) { for (let i = 0; i < rows.length; i++) { const r = rows[i].map(v => String(v || '').trim()); if (r.indexOf('نام کاربری') >= 0 || r.indexOf('کد پرسنلی') >= 0 || r.indexOf('username') >= 0) return i; } return -1; } function findColumn(headers, names) { for (let i = 0; i < names.length; i++) { const x = headers.indexOf(names[i]); if (x >= 0) return x; } return -1; } function isActiveStatus(status) { const s = normalizeDigits(status).trim().toLowerCase(); return s === '' || s === 'فعال' || s === 'active' || s === 'yes' || s === 'true' || s === '1'; } function normalizeDigits(v) { return String(v).replace(/[۰-۹]/g, d => '۰۱۲۳۴۵۶۷۸۹'.indexOf(d)).replace(/[٠-٩]/g, d => '٠١٢٣٤٥٦٧٨٩'.indexOf(d)); } function normalizeStoreCode(v) { return normalizeDigits(v || '').replace(/\s+/g, '').toUpperCase(); } function splitPersianList(v) { return String(v || '').split(/[،,]/).map(x => x.trim()).filter(Boolean); } function parseParts(v) { return splitPersianList(v).map(item => { const m = item.match(/^(.*?)\s*[×x]\s*(\d+)$/); return m ? { name: m[1].trim(), qty: Number(m[2]) || 1 } : { name: item, qty: 1 }; }); } function arrayToText(items) { return Array.isArray(items) ? items.filter(Boolean).join('، ') : ''; } function partsToText(parts) { return Array.isArray(parts) ? parts.filter(p => p && p.name).map(p => String(p.name).trim() + ' ×' + (Number(p.qty) || 1)).join('، ') : ''; } function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
